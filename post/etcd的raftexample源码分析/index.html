Etcd的raftexample源码分析

2020年5月19日
| etcd raft
| 阅读

etcd/raft是一个完成度很高的raft库，内部提供了一个raftexample的kv存储器，可以通过url对kv进行增加数据和查询数据，同时能对集群节点配置信息修改。

raftexample 源码分析

在了解raftexample之前，要先对底层的etcd/raft有一个初步的了解，etcd/raft是一个分布式一致性的库，其内部是一个状态机（也就是提供不同的输入，返回给你不同的输出），其主要提供了leader选举,log Entries等功能。但是像应用层的一些逻辑，WAL，Snapshot等都没有实现，而是暴露了接口，然用户根据自己的实际逻辑进行处理，这也给初学者带来了一定的困扰。

etcd/raft通过Node和上层应用进行交互，在raftexample/raft.go中对其进行封装和使用。

先看一下整体布局,主要包括五个文件。

我们跟随着使用存储服务的逻辑，从最简单的http处理开始看。可以看到其处理了四种HTTP方法，PUT表示插入数据，GET表示读取数据，POST表示增加节点，DELETE删除节点。可以看到PUT和GET分别是调用kvstore的Propose方法和Lookup方法，而POST和DELETE分别是新建了一个raftpb.ConfChange对象，然后传递到confChangeC管道。

// Handler for a http based key-value store backed by raft
type httpKVAPI struct {
    //store存储结构
    store       *kvstore
    confChangeC chan<- raftpb.ConfChange
}

//PUT插入数据，GET读取数据，POST增加节点，DELETE删除节点
func (h *httpKVAPI) ServeHTTP(w http.ResponseWriter, r *http.Request) {
    key := r.RequestURI
    switch {
    case r.Method == "PUT":
        v, err := ioutil.ReadAll(r.Body)
        ...
        h.store.Propose(key, string(v))

        w.WriteHeader(http.StatusNoContent)
    case r.Method == "GET":
        if v, ok := h.store.Lookup(key); ok {
            w.Write([]byte(v))
        } else {
            http.Error(w, "Failed to GET", http.StatusNotFound)
        }
    case r.Method == "POST":
        url, err := ioutil.ReadAll(r.Body)
        ...
        nodeId, err := strconv.ParseUint(key[1:], 0, 64)
        ...

        cc := raftpb.ConfChange{
            Type:    raftpb.ConfChangeAddNode,
            NodeID:  nodeId,
            Context: url,
        }
        //配置变化传到confChangC中
        h.confChangeC <- cc

        w.WriteHeader(http.StatusNoContent)
    case r.Method == "DELETE":
        nodeId, err := strconv.ParseUint(key[1:], 0, 64)
        ...

        cc := raftpb.ConfChange{
            Type:   raftpb.ConfChangeRemoveNode,
            NodeID: nodeId,
        }
        h.confChangeC <- cc

        w.WriteHeader(http.StatusNoContent)
    default:
        w.Header().Set("Allow", "PUT")
        w.Header().Add("Allow", "GET")
        w.Header().Add("Allow", "POST")
        w.Header().Add("Allow", "DELETE")
        http.Error(w, "Method not allowed", http.StatusMethodNotAllowed)
    }
}

接下来我们分开来看，首先看一下kvstore。kvstore是一个应用层的存储，底层是etcd/raft库。Lookup方法直接从kvStore中取键值对，Propose方法则是先编码，然后传入proposeC管道。除此之外，在newKVStore新建存储中，开了一个协程执行readCommits，readCommits的功能主要是监听着commitC管道，一旦有其他协程往管道中传输已提交（raft库已commited）的信息，则立刻解码并插入数据到kvStore

// a key-value store backed by raft
type kvstore struct {
	proposeC    chan<- string       // channel for proposing updates
	mu          sync.RWMutex
	kvStore     map[string]string   // current committed key-value pairs
	snapshotter *snap.Snapshotter
}

type kv struct {
	Key string
	Val string
}

func newKVStore(snapshotter *snap.Snapshotter, proposeC chan<- string, commitC <-chan *string, errorC <-chan error) *kvstore {
	s := &kvstore{proposeC: proposeC, kvStore: make(map[string]string), snapshotter: snapshotter}
	// replay log into key-value map
	s.readCommits(commitC, errorC)
	// read commits from raft into kvStore map until error
	go s.readCommits(commitC, errorC)
	return s
}

//kv查询
func (s *kvstore) Lookup(key string) (string, bool) {
	s.mu.RLock()
	v, ok := s.kvStore[key]
	s.mu.RUnlock()
	return v, ok
}

//kv插入，先编码然后传入proposeC的管道
func (s *kvstore) Propose(k string, v string) {
	var buf bytes.Buffer
	if err := gob.NewEncoder(&buf).Encode(kv{k, v}); err != nil {
		log.Fatal(err)
	}
	s.proposeC <- buf.String()
}

func (s *kvstore) readCommits(commitC <-chan *string, errorC <-chan error) {
	for data := range commitC {
		if data == nil {
			// done replaying log; new data incoming
			// OR signaled to load snapshot
			snapshot, err := s.snapshotter.Load()
			if err == snap.ErrNoSnapshot {
				return
			}
			if err != nil && err != snap.ErrNoSnapshot {
				log.Panic(err)
			}
			log.Printf("loading snapshot at term %d and index %d", snapshot.Metadata.Term, snapshot.Metadata.Index)
			if err := s.recoverFromSnapshot(snapshot.Data); err != nil {
				log.Panic(err)
			}
			continue
		}

		var dataKv kv
		dec := gob.NewDecoder(bytes.NewBufferString(*data))
		if err := dec.Decode(&dataKv); err != nil {
			log.Fatalf("raftexample: could not decode message (%v)", err)
		}
		s.mu.Lock()
		s.kvStore[dataKv.Key] = dataKv.Val
		s.mu.Unlock()
	}
	if err, ok := <-errorC; ok {
		log.Fatal(err)
	}
}

再看一下另一边，POST和DELETE方法中的confChangeC是哪里在监听管道呢?其实是在raft.go中，通过newRaftNode -> startRaft -> severChannels,这一部分核心代码很重要。serveChannels里面开了两个协程，一个协程用于监听proposeC（kvstore.Propose方法传入的数据）和confChangeC（httpKVAPI.ServeHTTP中POST和DELETE传入的数据），然后通过raftNode.node接口传数据到raft底层进行节点之间的同步处理。另一协程用于监听从底层raft传回的Ready数据，应用负责写WAL和网络传输发送给集群其他节点，在publishEntries将commited的数据传给你commitC（kvstore.readCommits负责监听管道），最后通知底层raft状态机应用更新完毕，可以准备下一次Ready了。

func (rc *raftNode) serveChannels() {
	snap, err := rc.raftStorage.Snapshot()
	if err != nil {
		panic(err)
	}
	rc.confState = snap.Metadata.ConfState
	rc.snapshotIndex = snap.Metadata.Index
	rc.appliedIndex = snap.Metadata.Index

	defer rc.wal.Close()

	ticker := time.NewTicker(100 * time.Millisecond)
	defer ticker.Stop()

	// send proposals over raft
	go func() {
		var confChangeCount uint64 = 0

		//监听proposeC和confChanegC，处理http中发来的请求
		for rc.proposeC != nil && rc.confChangeC != nil {
			select {
			case prop, ok := <-rc.proposeC:
				if !ok {
					rc.proposeC = nil
				} else {
					// blocks until accepted by raft state machine
					rc.node.Propose(context.TODO(), []byte(prop))
				}

			case cc, ok := <-rc.confChangeC:
				if !ok {
					rc.confChangeC = nil
				} else {
					confChangeCount += 1
					cc.ID = confChangeCount
					rc.node.ProposeConfChange(context.TODO(), cc)
				}
			}
		}
		// client closed channel; shutdown raft if not already
		close(rc.stopc)
	}()

	// event loop on raft state machine updates
	for {
		select {
		case <-ticker.C:
			rc.node.Tick()

		// store raft entries to wal, then publish over commit channel
		case rd := <-rc.node.Ready():

			//底层raft库通过Ready组件传给应用，应用负责写WAL和网络传输发送给集群其他节点
			//写WAL
			rc.wal.Save(rd.HardState, rd.Entries)
			if !raft.IsEmptySnap(rd.Snapshot) {
				rc.saveSnap(rd.Snapshot)
				rc.raftStorage.ApplySnapshot(rd.Snapshot)
				rc.publishSnapshot(rd.Snapshot)
			}
			rc.raftStorage.Append(rd.Entries)
			//发送给Follower
			rc.transport.Send(rd.Messages)
			//当底层raft组件判定请求在集群多数节点已经完成状态复制后，raft会将其视为可commit，
			//raft组件会将一个commit消息通过commit channel返回给应用(kvstore)，
			//应用在收到该commit消息后，将其应用至应用状态机，即存储在内存的kv map之中
			//将已经conmmited的日志提交到状态机
			if ok := rc.publishEntries(rc.entriesToApply(rd.CommittedEntries)); !ok {
				rc.stop()
				return
			}
			rc.maybeTriggerSnapshot()
			//通知底层raft状态机应用更新完毕，可以准备下一次Ready了
			rc.node.Advance()

		case err := <-rc.transport.ErrorC:
			rc.writeError(err)
			return

		case <-rc.stopc:
			rc.stop()
			return
		}
	}
}

基本上流程到这里就很清晰了，总结一下etcd/raft库的使用：etcd/raft底层只是对raft核心做了封装，并给定了数据结构和接口供应用层使用，应用层需要做许多自己的处理（包括WAL，Snapshot，节点之间传输，包括自己的功能如kv存储）才能正在使用起来整个库。
